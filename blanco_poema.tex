\begin{flushright}
	\textit{Recuerde el alma dormida,\\       
	avive el seso y despierte\\
	contemplando\\
	cómo se pasa la vida,\\
	cómo se viene la muerte\\              
	tan callando,\\
	cuán presto se va el placer,\\
	cómo, después de acordado,\\
	da dolor;\\
	cómo, a nuestro parecer,\\
	cualquiera tiempo pasado\\
	fue mejor.\\}
\end{flushright}
